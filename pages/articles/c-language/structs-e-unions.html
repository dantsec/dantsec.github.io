char c; }; // já o mesmo union teria { int(4) } => 4 bytes // o UNION pega o sizeof de maior tamanho dentre as variáveis, assim, todos os // membros compartilham da mesma memória union example { int a; int b; char c; }; // de resto, tudo é igual. ``` #### Structs ```c // struct structName { ... } surname; struct example { unsigned char example_c; unsigned int example_i; }; int main() { // struct example e; struct example e[10]; /* para zerarmos um struct: memset(&e, 0, sizeof(struct example)) */ memset(&e, 0, sizeof(struct example)) e[0].example_c = 'a'; e[0].example_i = 10; /* outras formas de inicializarmos essas variáveis: - os itens omitidos são zerados automaticamente!! // example_c = 'a' e example_i = 10 struct example e = { 'a', 10 }; // example_c = 'a' e example_i = 0 struct example e = { 'a' }; // example_c = 0 e example_i = 10 struct example e = { .num = 10 }; - podemos fazer uma atribuição composta (fora da declaração): struct example e = (struct example){ 'a', 10 }; */ // tamanho do struct printf("sizeof(struct example): %zu", sizeof(struct example)); return 0; } ``` - Typedef - Serve para definir / redefinir tipos ```c // usamos o typedef para fazer um "alias" para os tipos de dados typedef struct Distance { int feet;
